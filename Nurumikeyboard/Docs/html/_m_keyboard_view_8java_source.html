533  public void checkDirection(PtIdLinkedWithPtIndex pp, PointF pt)
534  {
535  CircleLinkedWithPtId cp = new CircleLinkedWithPtId();
536  for(int i=0; i<numFingers; i++)
537  {
538  if(clp.size() <= i)
539  return;
540  cp = clp.get(i);
541  if(pp.pointerId == cp.pointerId)
542  break;
543  }
544 
545  PointF oldPt = new PointF();
546  int circleNum = -1;
547  for(int i=0; i<numFingers; i++)
548  {
549  if(oldPtArr.size() <= i)
550  break;
551  oldPt = oldPtArr.get(i);
552  circleNum = checkTouchedCircle((int)oldPt.x, (int)oldPt.y);
553  if(cp.circleNum == circleNum)
554  break;
555  }
556  if(circleNum == -1) return;
557 
558  float distanceX = oldPt.x - pt.x;
559  float distanceY = oldPt.y - pt.y;
560 
561  if( Math.abs(distanceX) < SWIPE_MIN_DISTANCE && Math.abs(distanceY) < SWIPE_MIN_DISTANCE )
562  motion[circleNum-1] = DIRECTION_DOT;
563  else if( Math.abs(distanceX) > SWIPE_MIN_DISTANCE && Math.abs(distanceY) > SWIPE_MIN_DISTANCE )
564  motion[circleNum-1] = INVALID_DIRECTION;
565  else if( Math.abs(distanceX) > SWIPE_MIN_DISTANCE && distanceX > 0)
566  motion[circleNum-1] = DIRECTION_LEFT;
567  else if( Math.abs(distanceX) > SWIPE_MIN_DISTANCE && distanceX < 0)
568  motion[circleNum-1] = DIRECTION_RIGHT;
569  else if( Math.abs(distanceY) > SWIPE_MIN_DISTANCE && distanceY < 0)
570  motion[circleNum-1] = DIRECTION_DOWN;
571  else if( Math.abs(distanceY) > SWIPE_MIN_DISTANCE && distanceY > 0)
572  motion[circleNum-1] = DIRECTION_UP;
573  else
574  motion[circleNum-1] = INVALID_DIRECTION;
575  } // checkDirection fin
