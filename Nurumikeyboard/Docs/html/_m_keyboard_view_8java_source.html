556  public void checkDirection(PtIdLinkedWithPtIndex pp, PointF pt)
557  {
558  CircleLinkedWithPtId cp = new CircleLinkedWithPtId();
559  for(int i=0; i<numFingers; i++)
560  {
561  if(clp.size() <= i)
562  return;
563  cp = clp.get(i);
564  if(pp.pointerId == cp.pointerId)
565  break;
566  }
567 
568  PointF oldPt = new PointF();
569  int circleNum = -1;
570  for(int i=0; i<numFingers; i++)
571  {
572  if(oldPtArr.size() <= i)
573  break;
574  oldPt = oldPtArr.get(i);
575  circleNum = checkTouchedCircle((int)oldPt.x, (int)oldPt.y);
576  if(cp.circleNum == circleNum)
577  break;
578  }
579  if(circleNum == -1) return;
580 
581  float distanceX = oldPt.x - pt.x;
582  float distanceY = oldPt.y - pt.y;
583 
584  if( Math.abs(distanceX) < SWIPE_MIN_DISTANCE && Math.abs(distanceY) < SWIPE_MIN_DISTANCE )
585  motion[circleNum-1] = DIRECTION_DOT;
586  else if( Math.abs(distanceX) > SWIPE_MIN_DISTANCE && Math.abs(distanceY) > SWIPE_MIN_DISTANCE )
587  motion[circleNum-1] = INVALID_DIRECTION;
588  else if( Math.abs(distanceX) > SWIPE_MIN_DISTANCE && distanceX > 0)
589  motion[circleNum-1] = DIRECTION_LEFT;
590  else if( Math.abs(distanceX) > SWIPE_MIN_DISTANCE && distanceX < 0)
591  motion[circleNum-1] = DIRECTION_RIGHT;
592  else if( Math.abs(distanceY) > SWIPE_MIN_DISTANCE && distanceY < 0)
593  motion[circleNum-1] = DIRECTION_DOWN;
594  else if( Math.abs(distanceY) > SWIPE_MIN_DISTANCE && distanceY > 0)
595  motion[circleNum-1] = DIRECTION_UP;
596  else
597  motion[circleNum-1] = INVALID_DIRECTION;
598  } // checkDirection fin
